a) How many orders were shipped by Speedy Express in total?

SELECT 
COUNT (OrderID)
FROM ORDERS
INNER JOIN Shippers on Orders.ShipperID = Shippers.ShipperID
WHERE ShipperName like "Speedy Express";

ANS: 54



b) What is the last name of the employee with the most orders?


SELECT
E.LASTNAME,
MAX(Y.ORDERCOUNT)
FROM Employees AS E JOIN(
	SELECT 
	EMPLOYEEID,
	COUNT(ORDERID) AS ORDERCOUNT
	FROM Orders
	GROUP BY EMPLOYEEID) AS Y
	ON Y.EMPLOYEEID = E.EMPLOYEEID;

ANS:

c) What product was ordered the most by customers in Germany?

GET CUSTOMER IDS FOR GERMANS
GET PRODUCTS GERMANS ORDERED
GET MAX QUANITY



SELECT
PRODUCTID,
MAX(SUM)
FROM(
SELECT 
OD.PRODUCTID,
SUM(OD.QUANTITY) AS SUM
FROM ORDERDETAILS AS OD
INNER JOIN( --JOIN ORDER DETAILS AND CUSTOMERID ORDERID
	SELECT 
	O.ORDERID,
	Y.CUSTOMERID
	FROM ORDERS AS O
		INNER JOIN( --JOIN CUSTOMER TABLE WITH ORDER TABLE
		SELECT --THIS GETS US THE CUSTOMER IDS FOR GERMANS
		CUSTOMERID
		FROM CUSTOMERS
		WHERE COUNTRY LIKE "GERMANY") AS Y ON O.CUSTOMERID = Y.CUSTOMERID) AS X
    ON OD.ORDERID = X.ORDERID
    GROUP BY PRODUCTID);
